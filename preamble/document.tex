% Question Parts environment
\newenvironment{questionparts}[1][1]%
{
  \begin{enumerate}[\bfseries (i)]%
    \setcounter{enumii}{#1}
    \addtocounter{enumii}{-1}
    \setlength{\itemsep}{5mm}
    \setlength{\parskip}{8pt}
}
{
  \end{enumerate}
}

}
{

}
